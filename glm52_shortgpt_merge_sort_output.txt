Iterative merge sort works by dividing the array into subarrays of size 1, then merging them, then size 2, etc.

```python
def merge_sort(arr):
    n = len(arr)
    if n <= 1:
        return arr

    # Create a temporary array for merging
    temp = [0] * n
    width = 1
    while width < n:
        left = 0
        while left < n:
            # Calculate left and right subarrays
            left_idx = left
            left_end = min(left + width, n)  # end of left subarray
            right = min(left + width, n)      # start of right subarray
            right_end = min(left + 2 * width, n) # end of right subarray

            # Merge left and right subarrays
            i = left_idx
            l_ptr = left_idx
            r_ptr = right
            while l_ptr < left_end and r_ptr < right_end:
                if arr[l_ptr] <= arr[r_ptr]:
                    temp[i] = arr[l_ptr]
                    l_ptr += 1
                else:
                    temp[i] = arr[r_ptr]
                    r_ptr += 1
                i += 1

            # Copy remaining elements
            while l_ptr < left_end:
                temp[i] = arr[l_ptr]
                l_ptr += 1
                i += 1
            while r_ptr < right_end:
                temp[i] = arr[r_ptr]
                r_ptr += 1
                i += 1

            # Copy back to arr
            for j in range(left, right_end):
                arr[j] = temp[j]

            left += 2 * width
        width *= 2

    return arr
```
